**Цель:** Создать устойчивый поток ищущих людей, которые через бесплатный контент о камнях приходят к осознанию потребности в глубокой помощи, и мягко направить их к личному контакту с мастером.

**Бюджет:** 0 рублей.  
**Ресурсы:** Сайт magic-stone.org, Telegram-канал [Magic_of_stone](https://t.me/Magic_ofstone), Telegram-бот [@The_magic_of_stones_bot](https://t.me/The_magic_of_stones_bot), личное время.

**Принцип:** Три круга доступа (внешний, средний, внутренний). Всё явное — о камнях. Всё глубокое — только для посвящённых.

const TelegramBot = require('node-telegram-bot-api');
require('dotenv').config();

const token = process.env.BOT_TOKEN;
const bot = new TelegramBot(token, { polling: true });

// Хранилище состояний пользователей (в реальном проекте — база данных)
const userState = {};

// Стартовое сообщение с разветвлением
bot.onText(/\/start(?: (.+))?/, (msg, match) => {
  const chatId = msg.chat.id;
  const param = match[1]; // то, что после ?start=

  if (param === 'silence_path' || param === 'deep_request') {
    // Пришёл со скрытой страницы или по спецссылке
    bot.sendMessage(chatId,
      '🌑 *Тишина приветствует тебя.*\n\n' +
      'Я — проводник мастера. Здесь можно говорить прямо.\n' +
      'Расскажи, что привело тебя. Я слушаю.',
      { parse_mode: 'Markdown' }
    );
    userState[chatId] = { step: 'awaiting_deep_story' };
  } else if (param) {
    // Пришёл за конкретным камнем (например, ?start=aquamarine)
    sendStoneInfo(chatId, param);
  } else {
    // Обычный старт
    bot.sendMessage(chatId,
      '🪨 *Добро пожаловать в мир тихих камней.*\n\n' +
      'Я могу рассказать о любом минерале или помочь подобрать камень под ваш запрос.\n\n' +
      'Используйте меню 👇 или просто напишите название камня.',
      {
        parse_mode: 'Markdown',
        reply_markup: {
          keyboard: [
            ['📿 Узнать о камне', '🆘 Подобрать камень'],
            ['🌙 Мудрость дня', '🕯️ Глубина']
          ],
          resize_keyboard: true
        }
      }
    );
  }
});

// Обработка кнопок меню
bot.on('message', (msg) => {
  const chatId = msg.chat.id;
  const text = msg.text;
  const state = userState[chatId];

  // Если пользователь в режиме рассказа о глубинной проблеме
  if (state && state.step === 'awaiting_deep_story') {
    // Пересылаем сообщение мастеру (в личку)
    const masterChatId = process.env.MASTER_CHAT_ID; // Ваш ID в Telegram
    bot.forwardMessage(masterChatId, chatId, msg.message_id);
    bot.sendMessage(chatId,
      '🙏 *Благодарю за доверие.*\n' +
      'Твоя история услышана. Мастер свяжется с тобой лично в ближайшее время.\n' +
      'А пока — храни тишину внутри.',
      { parse_mode: 'Markdown' }
    );
    delete userState[chatId];
    return;
  }

  // Обработка кнопок
  if (text === '📿 Узнать о камне') {
    bot.sendMessage(chatId, 'Напиши название камня (например, "Аквамарин" или "Чёрный турмалин").');
  } else if (text === '🆘 Подобрать камень') {
    bot.sendMessage(chatId, 'Опиши свою ситуацию или чувство, которое хочешь изменить. Я подберу камень.');
  } else if (text === '🌙 Мудрость дня') {
    sendRandomWisdom(chatId);
  } else if (text === '🕯️ Глубина') {
    // Активация скрытого режима
    bot.sendMessage(chatId,
      '🕯️ *Ты коснулся глубины.*\n\n' +
      'Здесь нет спешки. Расскажи, что тревожит твою душу или тело. ' +
      'Я передам твои слова мастеру. Он прочитает и ответит тебе лично.',
      { parse_mode: 'Markdown' }
    );
    userState[chatId] = { step: 'awaiting_deep_story' };
  } else if (text && text.startsWith('/')) {
    // Обработка команд
    if (text === '/deep') {
      bot.sendMessage(chatId,
        '🌑 *Ты знаешь путь.*\n\nГовори, я слушаю.',
        { parse_mode: 'Markdown' }
      );
      userState[chatId] = { step: 'awaiting_deep_story' };
    }
  } else {
    // По умолчанию — пытаемся найти камень
    sendStoneInfo(chatId, text);
  }
});

// Функция отправки информации о камне
function sendStoneInfo(chatId, stoneName) {
  // Здесь может быть обращение к базе данных или статическому объекту
  // Для примера — заглушка
  const stones = {
    'аквамарин': '💧 *Аквамарин* — камень горловой чакры. Помогает выражать правду, снимает страх публичных выступлений. Хорошо носить у горла.',
    'чёрный турмалин': '🖤 *Чёрный турмалин* — мощный заземлитель и защитник. Ставьте у входа в дом или носите в левом кармане.',
    'обсидиан': '🌋 *Обсидиан* — зеркало души. Показывает скрытые травмы и родовые программы. Требует осторожного обращения.'
  };
  const key = stoneName.toLowerCase();
  if (stones[key]) {
    bot.sendMessage(chatId, stones[key], { parse_mode: 'Markdown' });
  } else {
    bot.sendMessage(chatId, '❓ Я пока не знаю такого камня. Попробуй другой или напиши "подобрать камень".');
  }
}

// Функция случайной мудрости
function sendRandomWisdom(chatId) {
  const wisdoms = [
    '🌙 "Тишина — это не отсутствие звука, а присутствие истины."',
    '💎 "Камень молчит, но говорит с тем, кто умеет слушать."',
    '🕊️ "Не ищи ответ вовне. Всё уже внутри тебя."'
  ];
  const random = wisdoms[Math.floor(Math.random() * wisdoms.length)];
  bot.sendMessage(chatId, random);
}

console.log('🤖 Бот запущен в тишине...');

### Сценарий 1: Обычный посетитель (камни)

1. Пользователь нажимает «Узнать о камне» или пишет название.
2. Бот выдаёт информацию.
3. В конце сообщения бот добавляет: *«Если хотите более личную рекомендацию — напишите "глубина"».*

### Сценарий 2: Ищущий (глубина)

1. Пользователь нажимает «🕯️ Глубина» или пишет `/deep`.
2. Бот переходит в режим сбора истории.
3. Всё, что пишет пользователь, пересылается мастеру в личку.
4. Бот отвечает: *«Мастер свяжется с вами лично»*.

### Сценарий 3: Переход со скрытой страницы сайта

1. Человек переходит по ссылке `?start=silence_path`.
2. Бот сразу приветствует его как «посвящённого» и предлагает рассказать о проблеме.
